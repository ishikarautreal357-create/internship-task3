<script>
  const QUESTIONS = [
    {
      type: 'single',
      q: 'Which HTML tag is used to link an external stylesheet?',
      options: ['<style>', '<link>', '<script>', '<css>'],
      answer: 1
    },
    {
      type: 'multi',
      q: 'Select all values that are valid CSS "display" properties.',
      options: ['flex', 'grid', 'inline-block', 'stack'],
      answer: [0, 1, 2]
    },
    {
      type: 'text',
      q: 'Fill in the blank: In JavaScript, the keyword used to declare a block-scoped variable that can be reassigned is ____.',
      accepted: ['let']
    },
    {
      type: 'single',
      q: 'Which of these is NOT a JavaScript data type?',
      options: ['Boolean', 'Undefined', 'Float', 'Symbol'],
      answer: 2
    },
    {
      type: 'multi',
      q: 'Which of these are valid ways to select an element in JavaScript?',
      options: ['document.getElementById()', 'document.querySelector()', 'document.findElement()', 'document.getElementsByClassName()'],
      answer: [0, 1, 3]
    },
    {
      type: 'text',
      q: 'Fill in the blank: The CSS property used to add space inside an element, between its content and border, is ____.',
      accepted: ['padding']
    }
  ];

  let current = 0;
  let userAnswers = new Array(QUESTIONS.length).fill(null);
  let selectedThisQ = null;

  const app = document.getElementById('app');

  // Escapes HTML special characters so option/question text (like "<script>")
  // renders as visible text instead of being parsed as actual HTML.
  function escapeHtml(str) {
    return String(str)
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#39;');
  }

  function init() {
    selectedThisQ = Array.isArray(userAnswers[current]) ? [...userAnswers[current]] : userAnswers[current];
    renderQuestion();
  }

  function renderQuestion() {
    const q = QUESTIONS[current];
    const progressPct = Math.round((current / QUESTIONS.length) * 100);
    const typeLabel = q.type === 'single' ? 'Single Select' : q.type === 'multi' ? 'Multi Select' : 'Fill in the Blank';

    let optionsHTML = '';

    if (q.type === 'single' || q.type === 'multi') {
      optionsHTML = `<div class="options">` + q.options.map((opt, i) => {
        const isSelected = q.type === 'single'
          ? selectedThisQ === i
          : Array.isArray(selectedThisQ) && selectedThisQ.includes(i);
        return `
          <div class="option ${q.type} ${isSelected ? 'selected' : ''}" onclick="selectOption(${i})">
            <span class="marker">${isSelected ? '✓' : ''}</span>
            <span>${escapeHtml(opt)}</span>
          </div>`;
      }).join('') + `</div>`;
    } else if (q.type === 'text') {
      const val = typeof selectedThisQ === 'string' ? selectedThisQ : '';
      optionsHTML = `
        <input type="text" class="blank-input" id="blankInput" placeholder="Type your answer..." value="${escapeHtml(val)}">
        <div class="hint">Not case-sensitive. Extra spaces are trimmed.</div>`;
    }

    app.innerHTML = `
      <div class="head">
        <div class="tag">Quiz &middot; Question ${current + 1} of ${QUESTIONS.length}</div>
        <h1>Knowledge Check</h1>
      </div>
      <div class="progress-row">
        <div class="progress-track"><div class="progress-fill" style="width:${progressPct}%"></div></div>
        <div class="progress-label">${progressPct}%</div>
      </div>
      <div class="body">
        <div class="qtype">${typeLabel}</div>
        <div class="question">${escapeHtml(q.q)}</div>
        ${optionsHTML}
        <div class="footer">
          <button class="nav-btn" onclick="prevQuestion()" ${current === 0 ? 'disabled' : ''}>Back</button>
          <button class="nav-btn primary" onclick="nextQuestion()">${current === QUESTIONS.length - 1 ? 'Finish' : 'Next'}</button>
        </div>
      </div>
    `;

    if (q.type === 'text') {
      const input = document.getElementById('blankInput');
      input.addEventListener('input', function () {
        selectedThisQ = this.value;
      });
      input.focus();
    }
  }

  function selectOption(i) {
    const q = QUESTIONS[current];
    if (q.type === 'single') {
      selectedThisQ = i;
    } else {
      if (!Array.isArray(selectedThisQ)) selectedThisQ = [];
      const idx = selectedThisQ.indexOf(i);
      if (idx > -1) selectedThisQ.splice(idx, 1);
      else selectedThisQ.push(i);
    }
    renderQuestion();
  }

  function saveCurrentAnswer() {
    const q = QUESTIONS[current];
    if (q.type === 'text') {
      const input = document.getElementById('blankInput');
      userAnswers[current] = input ? input.value.trim() : (selectedThisQ || '');
    } else {
      userAnswers[current] = selectedThisQ;
    }
  }

  function nextQuestion() {
    saveCurrentAnswer();
    if (current < QUESTIONS.length - 1) {
      current++;
      init();
    } else {
      showResults();
    }
  }

  function prevQuestion() {
    saveCurrentAnswer();
    if (current > 0) {
      current--;
      init();
    }
  }

  function isCorrect(q, ans) {
    if (ans === null || ans === undefined) return false;
    if (q.type === 'single') return ans === q.answer;
    if (q.type === 'multi') {
      if (!Array.isArray(ans)) return false;
      const a = [...ans].sort().join(',');
      const b = [...q.answer].sort().join(',');
      return a === b;
    }
    if (q.type === 'text') {
      return q.accepted.some(acc => acc.toLowerCase() === String(ans).toLowerCase().trim());
    }
    return false;
  }

  function formatUserAnswer(q, ans) {
    if (ans === null || ans === undefined || ans === '') return '(no answer)';
    if (q.type === 'single') return q.options[ans];
    if (q.type === 'multi') return Array.isArray(ans) && ans.length ? ans.map(i => q.options[i]).join(', ') : '(no answer)';
    return String(ans);
  }

  function formatCorrectAnswer(q) {
    if (q.type === 'single') return q.options[q.answer];
    if (q.type === 'multi') return q.answer.map(i => q.options[i]).join(', ');
    return q.accepted[0];
  }

  function showResults() {
    let score = 0;
    const reviewItems = QUESTIONS.map((q, i) => {
      const ans = userAnswers[i];
      const correct = isCorrect(q, ans);
      if (correct) score++;
      return `
        <div class="review-item ${correct ? 'correct' : 'incorrect'}">
          <div class="rq">${i + 1}. ${escapeHtml(q.q)}</div>
          <div class="ra your-answer ${correct ? '' : 'wrong'}">Your answer: ${escapeHtml(formatUserAnswer(q, ans))}</div>
          ${!correct ? `<div class="ra right-answer">Correct answer: ${escapeHtml(formatCorrectAnswer(q))}</div>` : ''}
        </div>`;
    }).join('');

    const total = QUESTIONS.length;
    const pct = Math.round((score / total) * 100);
    let verdict = 'Keep practicing — you\'ll get there.';
    if (pct === 100) verdict = 'Perfect score. Nothing to correct here.';
    else if (pct >= 75) verdict = 'Strong performance overall.';
    else if (pct >= 50) verdict = 'Decent attempt, a few gaps to close.';

    app.innerHTML = `
      <div class="results">
        <div class="stamp">
          <div class="score-num">${score}/${total}</div>
          <div class="score-den">SCORE</div>
        </div>
        <h2>Quiz Complete</h2>
        <p class="verdict">${verdict}</p>
        <div class="review-list">${reviewItems}</div>
        <button class="nav-btn primary" onclick="restart()">Try Again</button>
      </div>
    `;
  }

  function restart() {
    current = 0;
    userAnswers = new Array(QUESTIONS.length).fill(null);
    init();
  }

  init();
</script>
